\documentclass[a4paper,12pt]{article}

% No page numbers
\pagestyle{empty}

% Set margins
\usepackage[cm]{fullpage}

% Set paragraph indent to 2.2mm, even first
\usepackage{indentfirst}
\setlength{\parindent}{2.2mm}

% Image support
\usepackage{graphicx}

% Font Configuration

% Set font
\usepackage{fontspec}
\setmainfont[Numbers=OldStyle,
	     Scale=1.05,
	     ItalicFont={Adobe Garamond Pro Italic},
             BoldFont={Adobe Garamond Pro Semibold}
            ]{Adobe Garamond Pro}

% Set section to MidnightBlue with horizontal rule
\usepackage[dvipsnames,usenames]{color}
\usepackage[explicit]{titlesec}
\titleformat{\section}[display]{\Large\color{MidnightBlue}}{\thetitle}{1em}{#1}[{\titlerule}]

% Text formating

\newcommand{\standardHeader}[2]
{
	\begin{minipage}[b]{0.64\textwidth} #1 \end{minipage}
	\begin{minipage}[t]{0.32\textwidth} #2 \end{minipage}
}

\newcommand{\standardEntry}[2]
{
	\vspace{0.4\baselineskip}
	\begin{minipage}[t]{0.25\textwidth} \textsc{#1} \end{minipage}
	\begin{minipage}[t]{0.65\textwidth} #2 \end{minipage}
}

\newcommand{\stretchedEntry}[2]
{
	\standardEntry{#1}{#2}
}

\newcommand{\indentedEntry}[2]
{
	\standardEntry{#1}{#2}
}

\newcommand{\verticalSkip}[0]{\vspace{2.5mm}}

\newcommand{\headline}[1]{ {\font\1="Adobe Garamond Pro" at 20pt \1{#1}}}
